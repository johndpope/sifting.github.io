#define IMA_STEP_LEN 89
struct ADPCM_chan
{
    int32_t pred;
    int32_t ndx;
};
static inline int16_t
adpcm_expand (struct ADPCM_chan *restrict chan, uint8_t nibble)
{   /*http://www.cs.columbia.edu/~hgs/audio/dvi/IMA_ADPCM.pdf*/
    static int8_t ima_index_table[8] =
    {
        -1, -1, -1, -1, 2, 4, 6, 8
    };
    static int16_t ima_step_table[IMA_STEP_LEN] =
    { 
        7, 8, 9, 10, 11, 12, 13, 14, 16, 17, 
        19, 21, 23, 25, 28, 31, 34, 37, 41, 45, 
        50, 55, 60, 66, 73, 80, 88, 97, 107, 118, 
        130, 143, 157, 173, 190, 209, 230, 253, 279, 307,
        337, 371, 408, 449, 494, 544, 598, 658, 724, 796,
        876, 963, 1060, 1166, 1282, 1411, 1552, 1707, 1878, 2066, 
        2272, 2499, 2749, 3024, 3327, 3660, 4026, 4428, 4871, 5358,
        5894, 6484, 7132, 7845, 8630, 9493, 10442, 11487, 12635, 13899, 
        15289, 16818, 18500, 20350, 22385, 24623, 27086, 29794, 32767 
    };
    /*To recover the 16 bit sample we must do:
        for each nibble: 
            delta = step*(nibble + 1/2)/4 

    ...which is unfortunate left as is. However, if we expand some terms
    like so:

        step*(nibble + 1/2)/4 -> (step*nibble + step/2)/4 ->
        step*nibble/4 + step/8

    ...then all of a sudden we're dividing by nice powers of two. now,
    since there are only 4 bits of info, and one of them is dedicated to
    sign, that leaves only 3 bits of data. If we apply the function:

        f(x) = step*x/4

    ...for each bit set (b111 = 4|2|1) it expands to:

        f(4) + f(2) + f(1) -> 4*step/4 + 2*step/4 + 1*step/4 -> 
        step + step/2 + step/4

    ...which means we can do the multiplication through repeated addition.

    cool.
    */
    int32_t delta;
    int32_t step;
    /*Load the step*/
    if (chan->ndx < 0) chan->ndx = 0;
    else if (chan->ndx >= IMA_STEP_LEN) chan->ndx = IMA_STEP_LEN - 1;
    step = ima_step_table[chan->ndx];
    chan->ndx += ima_index_table[nibble&7];
    /*Recover the delta value*/
#if 0
    /*Reference implementation. Has precision issues. Don't use ever.*/
    delta = step>>3;                /*step/8 (= 1/2 in org. eq.)*/
    if (nibble&1) delta += step>>2; /*step/4*/
    if (nibble&2) delta += step>>1; /*step/2*/
    if (nibble&4) delta += step;    /*step*/
#elif 0 
    /*Improved version, uses fixed point to achieve parity with the
    reference. This version may work better on ARM style cpus where
    every instruction has an implicit condition*/
    delta = step>>1;                /*step/2 (= 1/2 in org. eq.)*/
    if (nibble&1) delta += step;    /*step*/
    if (nibble&2) delta += step<<1; /*step*2*/
    if (nibble&4) delta += step<<2; /*step*4*/
    delta >>= 2;                    /*shift out fractional bits*/
#elif 0
    /*Same as above, but with the branches eliminated through some bit tricks.
    Probably suboptimal on most modern machines, but could be great on cpus
    with expensive branching and multiplication*/
#define STEP(x, shift) do {
        register uint8_t tmp = x;
        delta += (step&((tmp<<16) - tmp))<<shift;
    } while (0);
    delta = step>>1;                /*step/2 (= 1/2 in org. eq.)*/
    STEP(nibble&1, 0);              /*step*/
    STEP(nibble&2, 1);              /*step*2*/
    STEP(nibble&4, 2);              /*step*4*/
    delta >>= 2;                    /*shift out fractional bits*/
#undef STEP
#else
    /*This version is best on modern cpus*/
    delta = step*(nibble&7) + (step>>1);
    delta >>= 2;
#endif
    /*Update predicted value and saturate*/
    if (nibble&8) chan->pred -= delta;
    else chan->pred += delta;
    if (chan->pred > INT16_MAX) chan->pred = INT16_MAX;
    else if (chan->pred < INT16_MIN) chan->pred = INT16_MIN;
    /*The predicted value is now the uncompressed sample*/
    return (int16_t)chan->pred;
}
static void
adpcm_decode (int16_t *res, uint8_t *data, size_t size, uint16_t chans)
{   /*Initialise local state*/
    uint16_t i;
    struct ADPCM_chan cs[2];
    for (i = 0; i < chans; i++)
    {/*First each channel has a 4 byte header at the start of the block.
    The first two bytes are the initial sample, followed by a byte index then
    a null padding byte. NB: the output takes the initial sample!*/
        cs[i].pred = res[0] = *(int16_t *)data;
        cs[i].ndx = data[2];
        data += 4;
        size -= 4;
        res++;
    }
    /*Recover the samples from the coded block. Like PCM samples are
    interleaved, but unlike PCM the samples are interleaved in groups of 8.
    So first 8 samples are left, then the next 8 are right. Very annoying.*/
    while (size)
    {
        uint8_t j, k;
        for (i = 0; i < chans; i++)
        {
            uint32_t stride = 0;
            for (j = 0; j < 4; j++)
            {/*Expand each group of samples. 8 samples = 4 bytes.*/
                uint8_t nibble = data[j];
                for (k = 0; k < 2; k++)
                {
                    res[stride + i] = adpcm_expand (cs + i, nibble);
                    stride += chans;
                    nibble >>= 4;
                }
            }
            size -= 4;
            data += 4;
        }
        /*Bump output up by 8 samples for each channel*/
        res += chans<<3;
    }
}
